```javascript
// 模板4：二叉树节点定义和构建
class TreeNode {
  constructor(val = 0, left = null, right = null) {
    this.val = val;
    this.left = left;
    this.right = right;
  }
}

// 从数组构建二叉树（华为常见输入格式）
function buildTree(arr) {
  if (!arr || arr.length === 0) return null;
  
  const root = new TreeNode(arr[0]);
  const queue = [root];
  let i = 1;
  
  while (queue.length && i < arr.length) {
    const node = queue.shift();
    
    // 左子节点
    if (arr[i] !== null && arr[i] !== undefined) {
      node.left = new TreeNode(arr[i]);
      queue.push(node.left);
    }
    i++;
    
    // 右子节点
    if (i < arr.length && arr[i] !== null && arr[i] !== undefined) {
      node.right = new TreeNode(arr[i]);
      queue.push(node.right);
    }
    i++;
  }
  
  return root;
}
```

```javascript
// 模板6：回溯算法通用模板
function backtrackTemplate(nums) {
  const result = [];
  const path = [];
  const used = new Array(nums.length).fill(false);
  
  function backtrack(depth) {
    // 终止条件
    if (depth === nums.length) {
      result.push([...path]); // 深拷贝
      return;
    }
    
    for (let i = 0; i < nums.length; i++) {
      // 剪枝条件（根据题目调整）
      if (used[i]) continue;
      if (i > 0 && nums[i] === nums[i-1] && !used[i-1]) continue;
      
      // 选择
      used[i] = true;
      path.push(nums[i]);
      
      // 递归
      backtrack(depth + 1);
      
      // 撤销选择
      path.pop();
      used[i] = false;
    }
  }
  
  backtrack(0);
  return result;
}
```

```javascript
function bfsTemplate(root) {
  if (!root) return [];
  
  const result = [];
  const queue = [root];
  
  while (queue.length > 0) {
    const levelSize = queue.length;
    const currentLevel = [];
    
    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift();
      currentLevel.push(node.val);
      
      // 根据题目扩展子节点
      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }
    
    result.push(currentLevel);
  }
  
  return result;
}
```

#### 1. 数组排序（哈希表、自定义排序）

这是考察JavaScript基础数据处理能力的典型题。

```javascript
const readline = require('readline');
const rl = readline.createInterface({ input: process.stdin, output: process.stdout });

function main(input) {
    // 输入示例："1,3,3,3,2,4,4,4,5"
    const nums = input.split(',').map(Number);
    const firstSeen = []; // 记录元素第一次出现的顺序
    const countMap = new Map(); // 统计元素出现次数

    for (const num of nums) {
        if (!countMap.has(num)) {
            firstSeen.push(num);
        }
        countMap.set(num, (countMap.get(num) || 0) + 1);
    }

    // 排序：先按次数降序，次数相同按首次出现顺序
    firstSeen.sort((a, b) => {
        const countDiff = countMap.get(b) - countMap.get(a);
        if (countDiff !== 0) return countDiff;
        // 由于firstSeen初始顺序就是首次出现顺序，所以无需额外索引
        return firstSeen.indexOf(a) - firstSeen.indexOf(b);
    });

    console.log(firstSeen.join(','));
}

rl.on('line', (line) => {
    main(line.trim());
});
```

#### 2. 贪吃的猴子（双指针/固定窗口）

这道题需要一点巧思，将“从两端取N次”转化为在数组中间找一个固定长度的最小窗口。

```javascript
const readline = require('readline');
const rl = readline.createInterface({ input: process.stdin, output: process.stdout });
let lines = [];
let totalLines = 3; // 根据题目，总共3行输入

rl.on('line', (line) => {
    lines.push(line);
    if (lines.length === totalLines) {
        const n = parseInt(lines[0]); // 数组长度
        const numbers = lines[1].split(' ').map(Number);
        const pickTimes = parseInt(lines[2]);

        const totalSum = numbers.reduce((a, b) => a + b, 0);
        // 如果取完所有香蕉
        if (pickTimes >= n) {
            console.log(totalSum);
            rl.close();
            return;
        }

        // 核心：问题等价于求一个长度为 (n - pickTimes) 的连续子数组的最小和
        const windowSize = n - pickTimes;
        let windowSum = numbers.slice(0, windowSize).reduce((a, b) => a + b, 0);
        let minWindowSum = windowSum;

        for (let i = windowSize; i < n; i++) {
            // 滑动窗口：减去左边，加上右边
            windowSum = windowSum - numbers[i - windowSize] + numbers[i];
            minWindowSum = Math.min(minWindowSum, windowSum);
        }

        // 总香蕉数减去这个最小窗口和，即为从两端能拿到的最大值
        console.log(totalSum - minWindowSum);
        rl.close();
    }
});
```

#### 3. 小华地图寻宝（DFS/BFS）

考察矩阵遍历，是DFS/BFS的模板题。

```javascript
const readline = require('readline');
const rl = readline.createInterface({ input: process.stdin, output: process.stdout });

function digitSum(num) {
    // 计算数位之和
    return num.toString().split('').reduce((sum, d) => sum + parseInt(d), 0);
}

rl.on('line', (line) => {
    const [m, n, k] = line.trim().split(' ').map(Number);
    const visited = Array.from({ length: m }, () => new Array(n).fill(false));
    let goldCount = 0;

    // 使用栈实现DFS
    const stack = [[0, 0]];
    visited[0][0] = true;

    // 方向数组
    const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];

    while (stack.length > 0) {
        const [x, y] = stack.pop();
        goldCount++; // 进入一个合法格子，获得黄金

        for (const [dx, dy] of dirs) {
            const newX = x + dx;
            const newY = y + dy;
            // 检查新坐标是否合法、未访问、且数位之和不大于k
            if (newX >= 0 && newX < m && newY >= 0 && newY < n &&
                !visited[newX][newY] &&
                (digitSum(newX) + digitSum(newY) <= k)) {
                visited[newX][newY] = true;
                stack.push([newX, newY]);
            }
        }
    }
    console.log(goldCount);
    rl.close();
});
```

## 题目2：字符串变换最小字符串

**描述**：交换一次字符使字符串字典序最小（不能交换相同位置）。

```javascript
const readline = require('readline');
const rl = readline.createInterface({ input: process.stdin, output: process.stdout });

rl.on('line', (line) => {
  let arr = line.trim().split('');
  const sorted = [...arr].sort();
  let found = false;
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] !== sorted[i]) {
      // 从后往前找到等于 sorted[i] 的最后一个位置
      let swapIndex = -1;
      for (let j = arr.length - 1; j > i; j--) {
        if (arr[j] === sorted[i]) {
          swapIndex = j;
          break;
        }
      }
      if (swapIndex !== -1) {
        [arr[i], arr[swapIndex]] = [arr[swapIndex], arr[i]];
        found = true;
        break;
      }
    }
  }
  console.log(arr.join(''));
});
```

```javascript
    let keySet = new Set();
    let cipher = [];
    // 密钥去重并保持顺序
    for (let ch of key.toUpperCase()) {
      if (!keySet.has(ch)) {
        keySet.add(ch);
        cipher.push(ch);
      }
    }
    // 补充剩余字母
    for (let ch of alphabet) {
      if (!keySet.has(ch)) {
        cipher.push(ch);
      }
    }
    // 加密
    let result = '';
    for (let ch of s.toUpperCase()) {
      if (ch >= 'A' && ch <= 'Z') {

      }
```

```javascript
    slow = slow.next;

      l2 = l2.next;

    cur = cur.next;

  return dummy.next;
}
```

```javascript
function buildTree(arr) {
  if (!arr || arr.length === 0 || arr[0] === null) return null;
  const root = new TreeNode(arr[0]);
  const queue = [root];
  let i = 1;
  while (queue.length && i < arr.length) {
    const node = queue.shift();
    if (arr[i] !== null && arr[i] !== undefined) {
      node.left = new TreeNode(arr[i]);
      queue.push(node.left);
    }
    i++;
    if (i < arr.length && arr[i] !== null && arr[i] !== undefined) {
      node.right = new TreeNode(arr[i]);
      queue.push(node.right);
    }
    i++;
  }
  return root;
}
```

  ```javascript
  const readline = require('readline');
  const rl = readline.createInterface({ input: process.stdin });
  let m, n, grid = [], starts = [];
  let lineNum = 0;
  rl.on('line', (line) => {
  if (lineNum === 0) {
    [m, n] = line.split(' ').map(Number);
  } else {
    const row = line.split(' ').map(Number);
    grid.push(row);
    for (let j = 0; j < n; j++) if (row[j] === 2) starts.push([lineNum-1, j]);
  }
  lineNum++;
  if (lineNum === m+1) {
    const vis = Array(m).fill().map(() => Array(n).fill(0));
    const dirs = [[-1,0],[1,0],[0,-1],[0,1]];
    const dfs = (r, c, seq) => {
      if (r < 0 || r >= m || c < 0 || c >= n || grid[r][c] === 1 || (vis[r][c] >> seq & 1)) return;

      }
    }

  }
  });
  ```

  ```javascript
  while (left < right) {
    const mid = Math.floor((left+right)/2);
    let machines = 1, cur = 0;
    for (let t of tasks) {
      if (cur + t <= mid) cur += t;
      else { machines++; cur = t; }
    }
    if (machines <= M) right = mid;
    else left = mid + 1;
  }
  console.log(left);
  });
  ```
- **解题思路**：二分最短时间，贪心检查是否能在M台机器内完成（将任务尽可能填满每台机器的时间）。

## 第24题：荒岛逃生游戏

- **核心考点**：BFS、迷宫
- **题目描述**：给定N×M网格，0路1障碍，从起点S到终点E，求最短步数，无法到达输出-1。
- **输入格式**：
  ```text
  3 3
  S0E
  010
  000
  ```
- **输出格式**：
  ```text
  4
  ```
- **JavaScript代码**：
  ```javascript
  const readline = require('readline');
  const rl = readline.createInterface({ input: process.stdin });
  let N, M, grid = [], start = null, end = null;
  let lineNum = 0;
  rl.on('line', (line) => {
  if (lineNum === 0) [N, M] = line.split(' ').map(Number);
  else {
    const row = line.split('');
    grid.push(row);
    for (let j=0; j<M; j++) {
      if (row[j] === 'S') start = [lineNum-1, j];
      if (row[j] === 'E') end = [lineNum-1, j];
    }
  }
  lineNum++;
  if (lineNum === N+1) {
    const queue = [[...start, 0]];
    const visited = Array(N).fill().map(() => Array(M).fill(false));
    visited[start[0]][start[1]] = true;
    const dirs = [[-1,0],[1,0],[0,-1],[0,1]];
    let ans = -1;
    while (queue.length) {
      const [x,y,step] = queue.shift();
      if (x === end[0] && y === end[1]) { ans = step; break; }
      for (let [dx,dy] of dirs) {
        const nx = x+dx, ny = y+dy;
        if (nx>=0 && nx<N && ny>=0 && ny<M && !visited[nx][ny] && grid[nx][ny] !== '1') {
          visited[nx][ny] = true;
          queue.push([nx,ny,step+1]);
        }
      }
    }
    console.log(ans);
    rl.close();
  }
  });
  ```
- **解题思路**：BFS求最短路径，注意起点和终点字符。

## 第25题：分苹果

- **核心考点**：数学、异或
- **题目描述**：A和B分一堆苹果，A希望两人分得的重量相等，B希望两人分得的重量异或值最大（即B的异或值最大）。已知苹果重量数组，求在B的异或值最大前提下，A得到的重量。若无法均分（异或总和不为0）则输出-1。
- **输入格式**：
  ```text
  3 5 6
  ```
- **输出格式**：
  ```text
  11
  ```
- **JavaScript代码**：
  ```javascript
  const readline = require('readline');
  const rl = readline.createInterface({ input: process.stdin });
  rl.on('line', (line) => {
  const nums = line.trim().split(' ').map(Number);
  let xorSum = 0;
  for (let num of nums) xorSum ^= num;
  if (xorSum !== 0) console.log(-1);
  else {
    nums.sort((a,b)=>a-b);
    let sum = nums.reduce((a,b)=>a+b,0);
    console.log(sum - nums[0]);
  }
  });
  ```
- **解题思路**：若总异或不为0则无法平分。否则，要使B的异或值最大，B应取所有苹果除最小重量外的所有苹果（因为异或总为0，A拿最小，B拿其余，B的异或等于最小重量，且这是最大值）。
