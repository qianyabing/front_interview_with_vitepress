#### 1. 二叉树的通俗理解

把二叉树类比成前端的「DOM树」：

- DOM树中每个节点有父节点、子节点；二叉树中每个节点最多只有**左子节点**和**右子节点**（「二叉」的由来）。

- 树的顶层节点叫「根节点」（对应`<html>`），没有子节点的节点叫「叶子节点」（对应`<span>`这类无子元素的标签）。

递归的核心逻辑：「先处理当前节点，再递归处理左子树，最后递归处理右子树」（不同遍历只是顺序变）。

// 1. 前序遍历（根 → 左 → 右）：机试最常考
function preorderTraversal(root) {
    const res = [];
    const traverse = (node) => {

        res.push(node.val); // 先存根节点值
        traverse(node.left); // 递归左子树
        traverse(node.right); // 递归右子树

    traverse(root);
    return res;
}

    const res = [];
    if (!root) return res;
    const queue = [root]; // 队列存当前层的节点
    while (queue.length) {
        const levelSize = queue.length; // 当前层的节点数
        const levelArr = [];

            const node = queue.shift(); // 取出队首节点

            // 左子节点入队

            // 右子节点入队
            if (node.right) queue.push(node.right);
        }
        res.push(levelArr);
    }
    return res;
}
